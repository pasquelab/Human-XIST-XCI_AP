Custom annotation files

Custom annotation files were build as follow:

Homo_sapiens.GRCh38.cdna.fa.gz and Homo_sapiens.GRCh38.ncrna.fa.gz were downloaded from ensembl.org and concatenated:

zcat Homo_sapiens.GRCh38.cdna.fa.gz Homo_sapiens.GRCh38.ncrna.fa.gz > Homo_sapiens.GRCh38.cdna.ncrna.fa.gz

This fasta file was used to generate an index for kallisto (0.46.0):

kallisto index -i kallisto_transcripts.idx Homo_sapiens.GRCh38.cdna.ncrna.fa.gz

The fasta file was also used to generate a table of all annotations:

zcat Homo_sapiens.GRCh38.cdna.ncrna.fa.gz | grep "^>" | cut -d " " -f 1,3,4 | grep -v 'scaffold' | sed 's/^>//'| sed 's/gene://' | sed 's/gene://' | sed 's/ /\t/g' > table_annot_allchr_simplified.tsv

kallisto (0.46.0) was run on bulk RNAseq data of Naive (Alfeghaly et al., 2023), TSC and EXMC cells (this study) using the following command line:

kallisto quant -i /shared/projects/xci/ReferenceGenomes/GRCh38/kallisto/Jeanne/kallisto_transcripts.idx -o ./ --rf-stranded --threads 8 -b 50 ../fastq/trimmed/PXGL_WT_R1_merged.fastq.gz ../fastq/trimmed/PXGL_WT_R2_merged.fastq.gz

annotate <- function(peaks_gr, annotations) {
  annotated_gr = annotate_regions(
    regions = peaks_gr,
    annotations = annotations,
    ignore.strand = TRUE,
    quiet = FALSE
  )
  return(data.frame(annotated_gr))
}

calculate_overlap <- function(data) {
  data <- data %>%
    mutate(intersection_start = ifelse(data$annot.start >= data$start, data$annot.start, data$start)) %>%
    mutate(intersection_end = ifelse(data$annot.end <= data$end, data$annot.end, data$end))
  
  data$overlapping_width <- data$intersection_end - data$intersection_start + 1 
  data$overlapping_percent <- data$overlapping_width/data$annot.width * 100
  
  return(data)
}

calculate_percentage <- function(data, gp_cell_type, control_df) {
  if (gp_cell_type == 'PXGL') {
    ctrl_gp <- filter(control_df, cell_type == 'whole_chr_PXGL') 
  } else if (gp_cell_type == 'TSC') {
    ctrl_gp <- filter(control_df, cell_type == 'whole_chr_TSC') 
  } else if (gp_cell_type == 'EXMC') {
    ctrl_gp <- filter(control_df, cell_type == 'whole_chr_EXMC') 
  }
  
  ctrl_gp <- ctrl_gp[,c('seqnames', 'annot.type', 'counts_annot')]
  
  data %>%
    left_join(ctrl_gp, by = c("seqnames", "annot.type"), suffix = c("", "_control")) %>%
    mutate(percentage_of_chr = counts_annot / counts_annot_control * 100) 
}


generate_histogram <- function(annotated_peaksets_list, mark, threshold_covered_percentage = 75, color_palette) {
  # preparing summary table
  annot_count_tables <- list() # initiating empty list
  for (i in 1:length(annotated_peaksets_list)) { 
    # counting nb of covered annotation, for each annot.type, for each peakset
    count_table <- annotated_peaksets_list[[i]] %>%
      filter(overlapping_percent > threshold_covered_percentage) %>%
      group_by(mark, cell_type, seqnames, annot.type) %>%
      summarise(counts_annot = n()) # counting the number of annotation falling in each category
    annot_count_tables[[i]] <- count_table # store the summary table in the list initiated above
  }
  # merging the list of df in one long df 
  annot_counts_alldata <- bind_rows(annot_count_tables)
  # removing all 'intergenic' data because unused here
  annot_counts_alldata <- annot_counts_alldata[!(annot_counts_alldata$annot.type == 'intergenic'),]
  # separating the long table in two, one ctrl, one with target data
  control_df <- annot_counts_alldata[ annot_counts_alldata$mark == 'None', ]
  target_df <- annot_counts_alldata[ annot_counts_alldata$mark == mark, ]
  # calculate percentage of control chr: 
  histogram_table <- target_df %>%
    group_by(cell_type) %>%
    group_modify(~ calculate_percentage(.,.y, control_df))
  # variables as factor to control order in the plot 
  histogram_table$cell_type <- factor(x = histogram_table$cell_type, levels = c("PXGL", "TSC", "EXMC"), labels = c("Naive", "TSC", "EXMC"))
  histogram_table$annot.type <- factor(x = histogram_table$annot.type, levels = c("intergenic", "enhancer", "upstream_1-5kb", "promoter", "gene_body"), label = c("intergenic", "enhancer", "upstream", "promoter", "gene body"))
  histogram_table$seqnames <- factor(x = histogram_table$seqnames, levels = c( 'chrX', 'chr7'))
  # plot
  ggplot(data = histogram_table, aes(x = annot.type, y = percentage_of_chr, fill = cell_type)) +
    # bars displaying the mean of each replicates
    geom_bar(stat = 'summary',position = "dodge", color='white') + 
    scale_fill_manual(values = color_palette) +
    # points for each replicates
    labs(
      title = paste("Percentage of chr annotations covered by", mark),
      subtitle = paste("Covered threshold = ", threshold_covered_percentage, "%"),
      x = "", 
      y = 'Percentage of annotations covered'
    ) +
    facet_grid(~ seqnames, scales = "free_x", space = "free_x") +
    scale_y_continuous(limits = c(0,50), expand = c(0,0)) +
    theme_classic() +
    theme(legend.title = element_blank())
}

getMostVariableGenes4<-function(counts,minCount=0.01,plot=TRUE,returnPlot=FALSE){
  counts<-counts[rowMeans(counts)>minCount,]
  dispTable<-data.frame(mu=rowMeans(counts),var=apply(counts,1,var),row.names =rownames(counts))
  dispTable$cv2 <- dispTable$var / dispTable$mu^2
  sumNullvariance<-sum(dispTable$cv2 <= 0)
  if(sumNullvariance>0){
    warning(paste0(sumNullvariance, " have null variance and will be removed"))
    dispTable<-dispTable[dispTable$cv2 > 0,]
  }
  fit<-loess(cv2~mu,data = log10(dispTable[,c("mu","cv2")]))
  dispTable$residuals<-fit$residuals
  dispTable$residuals2<-dispTable$residuals^2
  dispTable$fitted<-10^fit$fitted
  if(plot){
    require(ggplot2)
    require(ggrepel)
    require(circlize)
    g<-ggplot(dispTable,aes(x=mu,y=cv2,label=rownames(dispTable),fill=residuals))+
      geom_point(stroke=1/8,colour = "black",shape=21)+geom_line(aes(y=fitted),color="red",linewidth=1.5)+
      scale_x_log10()+scale_y_log10()
    if(returnPlot){
      return(g)
    }else{
      print(g)
    }
  }
  dispTable
}
